      }

        }

function prettyJSON(val, key, depth) {
 
  if (!depth) {
    depth = 0;
  }
 
  var prefix = '';

  }
 
  if (_.isArray(val) && !_.isObject(_.first(val))) {

      depth += 2;
    }

      prettyJSON(val, key, depth);
    });

  }
}
